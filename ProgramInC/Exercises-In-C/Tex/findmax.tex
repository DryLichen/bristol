\nsection{find\_max}
\begin{exercise}
\label{ex:findmax}
Write a program that finds the largest number entered by the user.
Executing the program will produce something like:
\begin{codesnippet}
How many numbers do you wish to enter ? 5
Enter 5 real numbers: 1.01 -3 2.2 7.0700 5
Maximum value: 7.07
\end{codesnippet}
\end{exercise}
